import matplotlib.pyplot as plt
data={"name":["mohmmad","ali","hasan","mahmmoud","kalid"],
     "age":[22,31,44,12,23],
     "salary":[290,340,200,500,420]}


plt.plot(data["salary"],data["name"],marker="o",linewidth=3,linestyle="--",color="blue")
plt.xlabel("Salary")
plt.ylabel("Name")
plt.title("Salary & Name")
plt.show()
